#!/bin/bash

while true; do
    echo "Выберите действие:"
    echo "1) Открыть nano"
    echo "2) Открыть vi"
    echo "3) Открыть браузер links"
    echo "4) Выйти"

    read choice
    case $choice in
        1) nano ;;
        2) vi ;;
        3) links ;;
        4) exit 0 ;;
        *) echo "Неверный ввод!";;
    esac
done
